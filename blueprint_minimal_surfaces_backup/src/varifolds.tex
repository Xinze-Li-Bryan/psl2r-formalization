\chapter{Varifolds}

\begin{definition}[Varifold]\label{def:varifold}
\lean{MinimalSurfaces.Varifold}
A finite nonnegative measure on the Grassmannian bundle of unoriented $2$-planes over $M$. \uses{}
\end{definition}

\begin{definition}[Integer varifold]\label{def:int-varifold}
\lean{MinimalSurfaces.IntegerVarifold}
An integer rectifiable varifold with multiplicity. \uses{}
\end{definition}

\begin{definition}[Mass]\label{def:mass}
\lean{MinimalSurfaces.mass}
The mass (area) of a varifold. \uses{}
\end{definition}

\begin{definition}[Varifold convergence]\label{def:varifold-conv}
\lean{MinimalSurfaces.convergesInVarifoldSense}
Weak convergence on the Grassmann bundle. \uses{}
\end{definition}
